message(STATUS "Creating build system for mtlnsolver")

set(CMAKE_Fortran_MODULE_DIRECTORY ${CMAKE_BINARY_DIR}/mod)

    "ngspice_interface.c"
)

	"network.F90"
	"network_manager.F90"
	"utils.F90"

	"dispersive.F90"
	"rational_approximation.F90"
	"circuit.F90"
	"ngspice_interface.F90"
	"preprocess.F90"
)

	${LAPACK_LIB}

	fhash
	ngspice_interface
	semba-types
